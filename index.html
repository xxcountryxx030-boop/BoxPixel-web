# ========================================= # BOXPIXEL BROWSER # Navegador simple en Python # ========================================= # INSTALAR: # pip install PyQt5 PyQtWebEngine import sys from PyQt5.QtWidgets import ( QApplication, QMainWindow, QWidget, QVBoxLayout, QHBoxLayout, QPushButton, QLineEdit, QLabel ) from PyQt5.QtCore import QUrl from PyQt5.QtGui import QFont from PyQt5.QtWebEngineWidgets import QWebEngineView class Navegador(QMainWindow): def __init__(self): super().__init__() self.setWindowTitle("BoxPixel Browser") self.resize(1300, 750) # ========================================= # NAVEGADOR WEB # ========================================= self.browser = QWebEngineView() self.browser.setUrl(QUrl("https://boxpixel-web.vercel.app/")) # ========================================= # LOGO TEXTO IZQUIERDA # ========================================= self.logo = QLabel("BoxPixel") self.logo.setFont(QFont("Arial", 18, QFont.Bold)) self.logo.setStyleSheet(""" color: #4285F4; padding-left: 10px; """) # ========================================= # BARRA URL / BÚSQUEDA # ========================================= self.url_bar = QLineEdit() self.url_bar.setPlaceholderText("Buscar o escribir URL...") self.url_bar.returnPressed.connect(self.navegar) self.url_bar.setStyleSheet(""" QLineEdit{ padding: 8px; border-radius: 10px; border: 1px solid gray; font-size: 14px; } """) # ========================================= # BOTONES # ========================================= self.btn_back = QPushButton("←") self.btn_forward = QPushButton("→") self.btn_reload = QPushButton("⟳") self.btn_back.clicked.connect(self.browser.back) self.btn_forward.clicked.connect(self.browser.forward) self.btn_reload.clicked.connect(self.browser.reload) # ========================================= # BARRA SUPERIOR # ========================================= top_bar = QHBoxLayout() top_bar.addWidget(self.logo) top_bar.addWidget(self.btn_back) top_bar.addWidget(self.btn_forward) top_bar.addWidget(self.btn_reload) top_bar.addWidget(self.url_bar) # ========================================= # LAYOUT PRINCIPAL # ========================================= layout = QVBoxLayout() layout.addLayout(top_bar) layout.addWidget(self.browser) container = QWidget() container.setLayout(layout) self.setCentralWidget(container) # Actualizar URL automáticamente self.browser.urlChanged.connect(self.actualizar_url) # ========================================= # NAVEGAR # ========================================= def navegar(self): texto = self.url_bar.text() # Si no tiene http -> buscar en Google if "." not in texto: url = f"https://www.google.com/search?q={texto}" else: if not texto.startswith("http"): texto = "https://" + texto url = texto self.browser.setUrl(QUrl(url)) # ========================================= # ACTUALIZAR URL # ========================================= def actualizar_url(self, q): self.url_bar.setText(q.toString()) # ========================================= # EJECUTAR APP # ========================================= app = QApplication(sys.argv) window = Navegador() window.show() sys.exit(app.exec_())
